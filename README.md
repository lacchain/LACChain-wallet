# Lacchain Wallet

## Requirements

- node Version: **14.17.0**

## Building

```shell
# credentials
REACT_APP_SUPPORTED_CHAIN_ID='0x9e55c'
REACT_APP_RPC_URL='https://writer-openprotest.lacnet.com'
REACT_APP_NODE_ADDRESS='0xad730de8c4bfc3d845f7ce851bcf2ea17c049585'
REACT_APP_SUPPORTED_PUBLIC_DIRECTORY_VERSION='1'
REACT_APP_ON_NOT_FOUND_PUB_KEY_RESOLUTION_METHOD=THROW_ON_NOT_FOUND_KEY_ERROR # Options: BLOCKCHAIN_TYPE1_POE, THROW_ON_NOT_FOUND_KEY_ERROR, CREDENTIAL_PROOF_TIME
REACT_APP_LAC1_DID_REGISTRY='0xb4FB2e9BB0001cc8eAAE528571915F35Cb74C864'
REACT_APP_LAC_DID_REGISTTRY='0xAB00e74C1b0A2313f552E869E6d55B5CdA31aFfe'
REACT_APP_LAC_DID_NETWORK_IDENTIFIER='openprotest'; # 'openprotest' or 'main
REACT_APP_MAILBOX_SERVICE='https://mailbox.openprotest.lacnet.com'
REACT_APP_MAILBOX_DID='did:lac:openprotest:0xf33bc23691245c2d5de99d7d45e9fdd113495870'
REACT_APP_DEFAULT_PUBLIC_DIRECTORY_TRUSTED_CONTRACT_ADDRESS='0xbBbfDe862725186E3D5332618a5546e60fcB75e6'
REACT_APP_PUBLIC_DIRECTORY_TRUSTED_CONTRACT_ADDRESSES='0xbBbfDe862725186E3D5332618a5546e60fcB75e6' # comma separated values
REACT_APP_DEFAULT_CHAIN_OF_TRUST_CONTRACT_ADDRESS='0xFE7CEF0D8E9A1dab4C4F57154e0191d67D4803AF'
REACT_APP_TRUSTED_CHAIN_OF_TRUST_CONTRACT_ADDRESSES='0xFE7CEF0D8E9A1dab4C4F57154e0191d67D4803AF' # comma separated values
REACT_APP_DEFAULT_VERIFICATION_REGISTRY_CONTRACT_ADDRESS='0x64CaA0fC7E0C1f051078da9525A31D00dB1F50eE'
REACT_APP_TRUSTED_VERIFICATION_REGISTRY_CONTRACT_ADDRESS='0x64CaA0fC7E0C1f051078da9525A31D00dB1F50eE' # comma separated values
REACT_APP_DID_METHOD=lac1 npm run build # specify the did method to be used on creating a new did through this application
```
